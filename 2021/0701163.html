视图集额外动作：

目的：给视图集添加额外的动作（方法）

1，子路由：

urlpatterns=[
    path('books/',views.BookInfoModelViewSet.as_view(&#123;'get':'list','post':'create'&#125;)),
    path('books/<int:pk>',views.BookInfoModelViewSet.as_view(&#123;'get': 'retrieve','put':'update','delete':'destroy'&#125;)),

    path('books/bread/',views.BookInfoModelViewSet.as_view(&#123;'get': 'bread_book'&#125;))
]

2，类视图：

#视图集
from rest_framework.response import Response
from rest_framework.viewsets import ModelViewSet

from booktest.models import BookInfo
from booktest.serializer import BookInfoModelSerializer

class BookInfoModelViewSet(ModelViewSet):
    #公共属性
    queryset = BookInfo.objects.all()
    serializer_class = BookInfoModelSerializer

    #1获取阅读量大于20的书籍
    def bread_book(self,request):
        #获取书籍
        books = BookInfo.objects.filter(bread__gt=20)
        #创建序列化器对象
        serializer = self.get_serializer(instance=books,many=True)

        return Response(serializer.data)

视图集额外动作 partial：

目的：给视图集添加额外的动作参数，并且更新局部信息
#1，子路由：
path('books/bread/<int:pk>',views.BookInfoModelViewSet.as_view(&#123;'put': 'update_book'&#125;))
#2，类视图：

class BookInfoModelViewSet(ModelViewSet):
    #公共属性
    queryset = BookInfo.objects.all()
    serializer_class = BookInfoModelSerializer
    ...

    #2，修改书籍编号为1，阅读量为99
    def update_book(self,request,pk):
        #获取参数
        book = self.get_object()
        data = request.data
        #创建序列化器
        serializer = self.get_serializer(instance = book,data=data,partial=True)
        #检验入库
        serializer.is_valid(raise_exception=True)
        serializer.save()
        #返回响应
        return Response(serializer.data,status=201)

路由router:

#目的：可以通过  DefaultRouter  和  SimpleRouter  两个类来自动生成路由
DefaultRouter  共有三对路由：

#列表路由
[
<URLPattern '^books/$' [name='hhh-list']>, 
<URLPattern '^books\.(?P<format>[a-z0-9]+)/?$' [name='hhh-list']>,
#详情路由
<URLPattern '^books/(?P<pk>[^/.]+)/$' [name='hhh-detail']>,
<URLPattern '^books/(?P<pk>[^/.]+)\.(?P<format>[a-z0-9]+)/?$' [name='hhh-detail']>, 
#根路由
<URLPattern '^$' [name='api-root']>,
<URLPattern '^\.(?P<format>[a-z0-9]+)/?$' [name='api-root']>
]

from rest_framework.routers import SimpleRouter,DefaultRouter
#创建路由对象
router = DefaultRouter()

#注册视图集
router.register('books',views.BookInfoModelViewSet,basename='hhh')
urlpatterns += router.urls
#输出结果
print(urlpatterns)

装饰 action:

可以通过 action 装饰方法，自动生成路由

类视图：

from rest_framework.decorators import action
class BookInfoModelViewSet(ModelViewSet):
    #公共属性
    ...
    #1获取阅读量大于20的书籍
    @action(methods=['GET'],detail=False)    #生成路由规则 ： 前缀/方法名/
    def bread_book(self,request):
        ...

    #2，修改书籍编号为1，阅读量为99
    @action(methods=['PUT'],detail=True)    #生成路由规则：前缀/&#123;pk&#125;/方法名/
    def update_book(self,request,pk):
        ...

生成的路由：

[
<URLPattern '^books/$' [name='hhh-list']>, 
<URLPattern '^books/bread_book/$' [name='hhh-bread-book']>, 
<URLPattern '^books/(?P<pk>[^/.]+)/$' [name='hhh-detail']>
]
